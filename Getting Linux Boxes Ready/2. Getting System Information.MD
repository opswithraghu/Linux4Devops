### Finding Linux Details
##### Until specified the commands work for both Ubuntu & Redhat
```bash
# Linux ersion
cat /etc/os-release

# Redhat only
cat /etc/redhat-release

# Kernel version
uname -r

# System Architecture
uname -m

# All system information
hostnamectl

# Number of CPU cores
nproc

# Detailed CPU info
lscpu

# Or
cat /proc/cpuinfo | grep "model name" | head -1

# Memory information
free -h

# Check available hardware info
sudo dmidecode -s system-product-name 2>/dev/null

# Quick details on Ubuntu
echo "OS: $(cat /etc/os-release)" && echo "Kernel: $(uname -r)" && echo "CPU: $(nproc) cores" && echo "Memory: $(free -h | grep Mem | awk '{print $2}')"

# Quick details on RHEL
echo "OS: $(cat /etc/redhat-release)" && echo "Kernel: $(uname -r)" && echo "CPU: $(nproc) cores" && echo "Memory: $(free -h | grep Mem | awk '{print $2}')"

# IP Address information
ip addr show
ip a

# Show all IP addresses
hostname -I

# Network Interfaces
ip link show

# Routing Table
ip route show

# DNS Configuration
cat /etc/resolv.conf

# Check Network Connectivity
ping -c 4 google.com

# Network Statistics
ss -tuln
netstat -tuln
# If netstat is unavailable, install on Ubuntu
sudo apt install net-tools -y
# If netstat is unavailable, install on Redhat
sudo dnf install net-tools -y
```
